# Giai đoạn 1: Sử dụng NVIDIA CUDA
FROM nvidia/cuda:11.6.1-runtime-ubuntu20.04 AS cuda

# Cài đặt các gói hệ thống cần thiết
RUN apt-get update && apt-get install -y \
    libgbm-dev \
    libgl1-mesa-glx \
    libxrender1 \
    && rm -rf /var/lib/apt/lists/*

# Thiết lập các biến môi trường cho CUDA
ENV PATH /usr/local/cuda/bin:$PATH
ENV LD_LIBRARY_PATH /usr/local/cuda/lib64:$LD_LIBRARY_PATH

# Giai đoạn 2: Cài đặt Python 3.6 và các gói phụ thuộc
FROM python:3.7 AS builder

# Cài đặt các gói hệ thống cần thiết
RUN apt-get update && apt-get install -y \
    build-essential \
    libgl1-mesa-glx \
    libxrender1 \
    libfontconfig1 \
    libglib2.0-0 \
    libsm6 \
    libxext6 \
    libxrender-dev \
    && rm -rf /var/lib/apt/lists/*

# Thiết lập thư mục làm việc
WORKDIR /parseq

# Sao chép các tệp cần thiết vào thư mục làm việc
COPY requirements.txt ./

# Cài đặt các gói phụ thuộc
RUN pip install --no-cache-dir -r requirements.txt

# Giai đoạn 3: Kết hợp CUDA và Python
FROM cuda AS final

# Thiết lập thư mục làm việc
WORKDIR /parseq

# Sao chép các gói đã cài đặt từ giai đoạn Python
COPY --from=builder /usr/local/lib/python3.6/dist-packages /usr/local/lib/python3.6/dist-packages
COPY --from=builder /parseq /parseq

# Mở cổng cho dịch vụ
EXPOSE 9001

# Khởi động ứng dụng
